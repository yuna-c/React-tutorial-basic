//Button.js
import React from 'react';
import classNames from 'classnames';
import './Button.scss';


// size : large, medium, small
// color : blue, pink, gray
function Button({ children, size, color, outline, fullWidth, className, ...rest }) { //props //outline, fullWidth : boolean = 배열형식으로 
    // console.log(rest);
    return (
        <>
            <button className={classNames(
                'Button', 
                size, 
                color, 
                {
                    outline,
                    fullWidth
                }, 
                className
                )}
                {...rest} // rest 안에 있는 모든것들을 rest안에 설정
            >
                {children}
            </button>
        </>
        // <button className={`Button ${size}`}>{children}</button>
    // "Button medium"
    )
}
Button.defaultProps = {
    size: 'medium',
    color: 'blue'
};

export default Button;



//Button.scss
// $ 변수선언
$blue: #339af0;
$gray: #495057;
$pink: #f06595;

@mixin button-color($color) {
    background: $color;
    &:hover {
        background: lighten($color, 10%);
    }

    &:active {
        background: darken($color, 10%);
    }

    &.outline{
       color: $color;
       background: transparent;
       border: 1px solid $color;
       &:hover{
           background: $color;
           color: #fff;
       }
    }
}

.Button {
    padding-left: 1rem;
    padding-right: 1rem;
    display: inline-flex;
    justify-content: center;
    align-items: center;
    border: none;
    border-radius: 4px; 
    font-weight: 700;
    color: #fff;
    outline: none;
    cursor: pointer;

    &.large {
        height:3rem;
        line-height:3rem;
        font-size: 1.25rem;
    }
    &.medium {
        height: 2.25rem;
        line-height: 2.25rem;
        font-size: 1rem;
    }
    &.small {
        height:1.75rem;
        line-height:1.75rem;
        font-size: 0.875rem;
    }

    // & : 자기 자신
    &.blue {
        @include button-color($blue);
    }
    &.gray {
        @include button-color($gray);
    }
    &.pink {
        @include button-color($pink);
    }
    

    // .Button + .Button{
    //     margin-left: 1rem; 
    // }
    & + & {
        margin-left: 1rem; 
    }
    &.fullWidth {
        width: 100%; 
        justify-content: center;
        & + & { 
            margin-left: 0;
            margin-top: 1rem;
        }
    }
}
